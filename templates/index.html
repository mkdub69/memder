    <script>

      let selected = null;


      function selectMeme(id) {

        document.getElementById("img1").classList.remove("selected");

        document.getElementById("img2").classList.remove("selected");

        document.getElementById(id).classList.add("selected");

        selected = id;

      }


      function submitForm(action) {

        if (!selected) {

          alert("Bitte wähle zuerst ein Meme aus.");

          return false;

        }

        const imgPath = selected === 'img1' ? "{{ img1 }}" : "{{ img2 }}";

        if (action === 'vote') {

          document.getElementById('voteInput').value = imgPath;

        } else {

          document.getElementById('flagInput').value = imgPath;

        }

        return true;

      }
